164 private void StartSimulationButton_Click(object sender, Avalonia.Interactivity.RoutedEventArgs e)
165 {
166 if (!_isRunning)
167 {
168 if (_scene.Balls.Count > 0)
169 {
170 var ball = _scene.Balls[0];
171 float speed = (float)_speedSlider.Value;
172
173 Random random = new Random();
174 float angle = (float)(random.NextDouble() * 2 * Math.PI);
175 Vector2 velocity = new Vector2(
176 (float)(Math.Cos(angle) * speed),
177 (float)(Math.Sin(angle) * speed)
178 );
179
180 ball.SetVelocity(velocity);
181
182 var world = _ecsManager.World;
183 var velocityPool = world.GetPool<BHS.Engine.ECS.Components.VelocityComponent>();
184 var sceneLinkPool = world.GetPool<BHS.Engine.ECS.Components.SceneObjectLink>();
185
186 var ballFilter = world.Filter<BHS.Engine.ECS.Components.VelocityComponent>()
187 .Inc<BHS.Engine.ECS.Components.SceneObjectLink>()
188 .End();
189
190 foreach (var entity in ballFilter)
191 {
192 ref var sceneLink = ref sceneLinkPool.Get(entity);
193 if (sceneLink.Reference == ball)
194 {
195 ref var velocityComponent = ref velocityPool.Get(entity);
196 velocityComponent.Velocity = velocity;
197 break;
198 }
199 }
200 }
201
202 _isRunning = true;
203 _gameTimer.Start();
204 }
205 }
