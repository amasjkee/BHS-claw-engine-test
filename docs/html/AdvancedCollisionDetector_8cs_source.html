205 private static bool CheckPointWallCollisionCCD(Ball ball, Vector2 wallPoint, Vector2 movementDirection, float movementLength, out Vector2 collisionPoint, out Vector2 normal, out float collisionTime)
206 {
207 collisionPoint = wallPoint;
208 normal = Vector2.Zero;
209 collisionTime = 0f;
210
211 float distanceToWall = Vector2.Distance(ball.Position, wallPoint);
212
213 if (distanceToWall <= ball.Radius)
214 {
215 collisionTime = 0f;
216 normal = (ball.Position - wallPoint).Normalized;
217 return true;
218 }
219
220 Vector2 directionToWall = wallPoint - ball.Position;
221 float dotProduct = Vector2.Dot(movementDirection.Normalized, directionToWall.Normalized);
222
223 if (dotProduct <= 0)
224 return false; // Шарик удаляется от стены
225
226
227 float a = movementDirection.SqrMagnitude;
228 float b = 2 * Vector2.Dot(ball.Position - wallPoint, movementDirection);
229 float c = Vector2.SqrDistance(ball.Position, wallPoint) - ball.Radius * ball.Radius;
230
231 float discriminant = b * b - 4 * a * c;
232 if (discriminant < 0)
233 return false;
234
235 float sqrtDiscriminant = MathHelper.Sqrt(discriminant);
236 float t1 = (-b - sqrtDiscriminant) / (2 * a);
237 float t2 = (-b + sqrtDiscriminant) / (2 * a);
238
239 float t = float.MaxValue;
240 if (t1 >= 0 && t1 <= 1)
241 t = MathHelper.Min(t, t1);
242 if (t2 >= 0 && t2 <= 1)
243 t = MathHelper.Min(t, t2);
244
245 if (t < float.MaxValue)
246 {
247 collisionTime = t;
248 Vector2 ballPosAtCollision = ball.Position + t * movementDirection;
249 normal = (ballPosAtCollision - wallPoint).Normalized;
250 return true;
251 }
252
253 return false;
254 }
